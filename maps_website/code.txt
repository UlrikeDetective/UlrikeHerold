<script>
    // Initialize the map
    const mapEurope = L.map('map').setView([50.5, -0.09], 4);
    
    // Add a simple base layer
    L.tileLayer('https://cartodb-basemaps-{s}.global.ssl.fastly.net/light_all/{z}/{x}/{y}.png', {
    attribution: '&copy; <a href="https://carto.com/attributions">CARTO</a>'
    }).addTo(mapEurope);

        // Function to generate random colors
        function getRandomColor() {
            const letters = '0123456789ABCDEF';
            let color = '#';
            for (let i = 0; i < 6; i++) {
                color += letters[Math.floor(Math.random() * 16)];
            }
            return color;
        }

        // Function to create a circle marker with a random color
        function createCircleMarker(lat, lng, popupText) {
            const randomColor = getRandomColor();
            const circleMarker = L.divIcon({
                className: 'custom-circle-marker',
                html: `<svg width="20" height="20" viewBox="0 0 20 20">
                            <circle cx="10" cy="10" r="9" fill="${randomColor}" />
                        </svg>`,
                iconSize: [20, 20],
                iconAnchor: [10, 10]
            });
            L.marker([lat, lng], { icon: circleMarker }).addTo(mapEurope)
                .bindPopup(popupText);
        }

        createCircleMarker(51.5, -0.09, "<b>London.</b> <br>Whats to say.<br> Been here 17 times so fare ü•∞");
        createCircleMarker(51.4545, -2.5879, "<b>Bristol</b><br>Let's see some streetart. üë®‚Äçüé®");


        </script>
